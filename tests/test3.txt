n = 5;

func countdown(x) {
    if (x > 0) {
        print(x);
        countdown(x - 1);
    }
}

func factorialPrint(x) {
    if (x == 0) {
        print(1);
    } else {
        factorialPrint(x - 1);
        print(x); 
    }
}

countdown(n);
factorialPrint(n);

while (n < 8) {
    print(n);
    n = n + 1;
}
